# This file contains a whitelist for the fields API for ingest scripts

class org.elasticsearch.script.field.Metadata {
    String getIndex()
    void setIndex(String)

    String getId()
    void setId(String)

    String getRouting()
    void setRouting(String)

    Long getVersion()
    void setVersion(Long)

    VersionType getVersionType()
    void setVersionType(VersionType)

    Op getOp()
    void setOp(Op)

    ZonedDateTime getTimestamp()
}

class org.elasticsearch.script.field.Op {
  Op NOOP
  Op INDEX
  Op DELETE
  Op CREATE
  String getName()
  Op fromString(String)
}

class org.elasticsearch.index.VersionType {
  VersionType INTERNAL
  VersionType EXTERNAL
  VersionType EXTERNAL_GTE
}

class org.elasticsearch.script.IngestScript {
    Metadata meta()
}

class org.elasticsearch.script.UpdateScript {
    Metadata meta()
}
