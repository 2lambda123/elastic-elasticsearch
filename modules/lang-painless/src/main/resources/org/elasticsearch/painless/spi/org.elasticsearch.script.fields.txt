# The whitelist for the fields api

# API
class org.elasticsearch.script.Field {
  String getName()
  boolean isEmpty()
  List getValues()
}

class org.elasticsearch.script.BooleanField {
  String getName()
  boolean isEmpty()
  List getValues()
  boolean getValue(boolean)
}

class org.elasticsearch.script.DoubleField {
  String getName()
  boolean isEmpty()
  List getValues()
  double getValue(double)
}

class org.elasticsearch.script.LongField {
  String getName()
  boolean isEmpty()
  List getValues()
  long getValue(long)
}

class org.elasticsearch.script.NonPrimitiveField {
  String getName()
  boolean isEmpty()
  List getValues()
  Object getValue(Object)
}

}

class org.elasticsearch.script.StringSortScript @no_import {
}
